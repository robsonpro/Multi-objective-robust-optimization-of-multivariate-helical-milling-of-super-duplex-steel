Pa1.rmse = function(x, TT) { 
  
  E <- (b0 + t(x)%*%b + t(x)%*%B%*%x)
  V <- (mse + t(g + t(D)%*%x)%*%(g + t(D)%*%x))
  
  rmse <- sqrt((E - TT)^2 + V)
  return(rmse)
}

Pa2.rmse = function(x, TT) { 
  
  E <- (b02 + t(x)%*%b2 + t(x)%*%B2%*%x)
  V <- (mse2 + t(g2 + t(D2)%*%x)%*%(g2 + t(D2)%*%x))
  
  rmse <- sqrt((E - TT)^2 + V)
  return(rmse)
}

  E <- (b03 + t(x)%*%b3 + t(x)%*%B3%*%x)
